import java.util.Arrays;
import java.util.Comparator;
import java.util.List;

class Process {
int pid, arrivalTime, burstTime, waitingTime, turnaroundTime;
public int remainingTime;

Process(int pid, int at, int bt) {
this.pid = pid;
this.arrivalTime = at;
this.burstTime = bt;
}
}

public class FCFS {
public static void main(String[] args) {

//Change Input in this block |
List<Process> processes = Arrays.asList(
new Process(1, 0, 5),
new Process(2, 1, 3),
new Process(3, 2, 8)
);

processes.sort(Comparator.comparingInt(p -> p.arrivalTime));
int time = 0;

for (Process p : processes) {
time = Math.max(time, p.arrivalTime);
p.waitingTime = time - p.arrivalTime;
time += p.burstTime;
p.turnaroundTime = p.waitingTime + p.burstTime;
}

System.out.println("FCFS Scheduling:");
for (Process p : processes) {
System.out.println("P" + p.pid + " WT=" + p.waitingTime + " TT=" + p.turnaroundTime);
}
}
}
